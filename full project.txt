        // Function to set the selected service from the previous page
        function setServiceFromURL() {
            const params = new URLSearchParams(window.location.search);
            const service = params.get("service");

            if (service) {
                document.getElementById("selectedService").value = service;
            }
        }

        function confirmBooking() {
            let name = document.getElementById("customerName").value;
            let number = document.getElementById("customerNumber").value;
            let service = document.getElementById("selectedService").value;
            let additionalService = document.getElementById("additionalService").value;
            let date = document.getElementById("bookingDate").value;
            let time = document.getElementById("bookingTime").value;

            if (name === "" || number === "" || service === "" || date === "" || time === "") {
                alert("Please fill in all required fields.");
                return;
            }

            let finalService = additionalService ? service + " & " + additionalService : service;

            alert("Booking Confirmed!\n\nName: " + name + "\nNumber: " + number + "\nService: " + finalService + "\nDate: " + date + "\nTime: " + time);
        }

        // Load selected service when page opens
        setServiceFromURL();
